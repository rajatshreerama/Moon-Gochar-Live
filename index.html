    <script>

        // --- आपका बेस डेटा (30 Jan 2026, 11:14 AM IST के आसपास का) ---

        const startTime = new Date("2026-01-30T11:14:00").getTime();

        const startLagna = 7.3361; // मेष 07:20:10 (Decimal)

        const startMoon = 70.1586; // मिथुन 10:09:31 (60 + 10.1586)


        function toDMS(deg) {

            let d = Math.floor(deg % 30);

            let m = Math.floor((deg % 1) * 60);

            let s = Math.floor(((deg % 1 * 60) % 1) * 60);

            return `${d}:${m < 10 ? '0'+m : m}:${s < 10 ? '0'+s : s}`;

        }


        const rPos = { 1:[200,140], 2:[100,70], 3:[70,100], 4:[140,200], 5:[70,300], 6:[100,330], 7:[200,265], 8:[300,330], 9:[330,300], 10:[260,200], 11:[330,100], 12:[300,70] };


        function update() {

            const now = new Date();

            const diff = (now.getTime() - startTime) / 1000;


            // लग्न गति: 1 सेकंड में 0.004166 डिग्री

            let currentLTotal = startLagna + (diff * 0.004166);

            let lSign = (Math.floor(currentLTotal / 30) % 12) + 1;

            let lDeg = currentLTotal % 30;


            // चंद्रमा गति: 13.18 डिग्री प्रति दिन

            let currentMTotal = startMoon + (13.18 / 86400 * diff);

            let mSign = (Math.floor(currentMTotal / 30) % 12) + 1;

            let mHouse = (mSign - lSign + 1); if(mHouse <= 0) mHouse += 12;


            let html = "";

            for(let i=1; i<=12; i++) {

                let rNum = (lSign + i - 2) % 12 + 1;

                html += `<text x="${rPos[i][0]}" y="${rPos[i][1]}" class="r-num">${rNum}</text>`;

                

                if(i === 1) {

                    html += `<text x="200" y="85" class="l-val">लग्न: ${toDMS(lDeg)}</text>`;

                }

                if(i === mHouse) {

                    html += `<text x="${rPos[i][0]}" y="${rPos[i][1]+30}" class="m-val">-चं ${toDMS(currentMTotal%30)}</text>`;

                }

            }

            document.getElementById('live-layer').innerHTML = html;

        }


        // हर 1 सेकंड में रिफ्रेश

        setInterval(update, 1000);

        update();

    </script>
